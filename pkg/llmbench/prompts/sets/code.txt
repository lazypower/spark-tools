    results := make([]Result, len(items))
    errCh := make(chan error)
    for i, item := range items {
        go func() {
            result, err := process(item)
            if err != nil {
                errCh <- err
                return
            }
            results[i] = result
        }()
    }
    select {
    case err := <-errCh:
        return nil, err
    default:
        return results, nil
    }
}
